### 2、普通的工具函数和hook的区别？
hook能够让在不使用class的情况下，使用class的一些特性，例如，副作用，状态管理等

function throttle(fn, time) {
   let timer ;
   const preTime = Date.now();
   return (...arg)=>{
       const currentTime = Date.now();
       const elipsTime = currentTime - preTime;
       if(elipsTime> time ){
           fn.apply(this,arg);
       }else{
           timer = setTimeout(()=>{
               fn.apply(this,arg)
           },elipsTime)
       }
   }

 }

function debounce(fn,time){
    let timer ;
    return (...arg)=>{
        if(timer){
            clearTimeout(timer);
            timer = null;
        }else{
            timer = setTimeout(()=>{
            fn.apply(this,arg)
        },time)
        }
    }
}
